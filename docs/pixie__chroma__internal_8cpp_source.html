872void PixieChroma::write_pix( char* message, int16_t x_dest, int16_t y_dest ){
873 int16_t offset_x = 0;
874 int16_t offset_y = 0;
875
876 just_wrapped = false;
877
878 uint8_t len = strlen( message );
879 for( uint8_t i = 0; i < len; i++ ){
880 if( message[i] == '\n' ){ // Newline, force line break
881 if(just_wrapped == true){
882 just_wrapped = false;
883 // Skip newline, auto wrapping just newline'd for us.
884 }
885 else{
886 x_dest = extra_space_left+display_padding_x;
887 offset_x = 0;
888 offset_y = display_height;
889 }
890 }
891 else if( line_wrap == true && x_dest+offset_x+display_width >= (matrix_width) ){ // End of line reached, wrap to new line if line_wrap enabled
892 add_char(
893 message[i],
894 x_dest + offset_x,
895 y_dest + offset_y
896 );
897 offset_x += display_width;
898
899 x_dest = extra_space_left+display_padding_x;
900 offset_x = 0;
901 offset_y = display_height;
902
903 just_wrapped = true;
904 }
905 else if( message[i] == 0 || message[i] == '\0' ){ // end of string
906 return;
907 }
908 else{ // Normal char
909 add_char(
910 message[i],
911 x_dest + offset_x,
912 y_dest + offset_y
913 );
914 offset_x += display_width;
915 }
916 }
917
918 cursor_x_temp = x_dest+offset_x;
919 cursor_y_temp = y_dest+offset_y;
920}

1701CRGB PixieChroma::kelvin_to_rgb( uint16_t temperature ){
1702 // Tanner Helland formula
1703 float _temperature;
1704 float _red;
1705 float _green;
1706 float _blue;
1707
1708 _temperature = constrain( temperature, 0, 65500 );
1709
1710 _red = _green = _blue = 0;
1711 float t = _temperature * 0.01;
1712
1713 if ( t <= 66 ){
1714 _red = 255;
1715 _green = ( 99.4708025861 * log( t ) ) - 161.1195681661;
1716 if ( t > 19 ){
1717 _blue = ( 138.5177312231 * log( t - 10 ) ) - 305.0447927307;
1718 }
1719 else{
1720 _blue = 0;
1721 }
1722 }
1723 else{
1724 _red = 329.698727466 * pow( t - 60, -0.1332047592 );
1725 _green = 288.1221695283 * pow( t - 60, -0.0755148492 );
1726 _blue = 255;
1727 }
1728
1729 float f = 0.01 * 100;
1730
1731 _red = constrain( f * _red, 0, 255 );
1732 _green = constrain( f * _green, 0, 255 );
1733 _blue = constrain( f * _blue, 0, 255 );
1734
1735 return CRGB( _red, _green, _blue );
1736}
